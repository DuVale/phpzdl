function dovoice(val) {
	var url = '';
	for ( rule in vrules ) {
		var re = new RegExp(rule,"ig");
		if (val.match(re)) url=vrules[rule];
	}
	if (url != '') parent.frames["main"].document.location.href=url;
	else           alert(val+': Command not recognized');
}
